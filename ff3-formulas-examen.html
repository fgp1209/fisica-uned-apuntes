FF3 — Fórmulas de examen

T1 — Cuántica vieja

\[E=hf=\frac{hc}{\lambda}\]

\[K_{\max}=hf-\phi\]

\[\lambda=\frac{h}{p}\]

\[\Delta x\Delta p\geq\frac{\hbar}{2}\]

\[\Delta\lambda=\frac{h}{mc}(1-\cos\theta)\]

T2 — Schrödinger

\[\psi_n(x)=\sqrt{\frac{2}{L}}\sin\left(\frac{n\pi x}{L}\right)\]

\[E_n=\frac{n^2h^2}{8mL^2}\]

\[P(a,b)=\int_a^b|\psi_n(x)|^2dx\]

\[T\simeq e^{-2\alpha a}\]

\[\alpha=\frac{\sqrt{2m(U_0-E)}}{\hbar}\]

T3 — Átomos

\[P(r)=4\pi r^2|\psi(r)|^2\]

\[\langle r^n\rangle=\int r^n|\psi|^2dV\]

\[E_n=-\frac{13,6\,\mathrm{eV}}{n^2}\]

\[\frac1\lambda=R\left(\frac1{n_f^2}-\frac1{n_i^2}\right)\]

T4 — Moléculas

\[\mu=\frac{m_1m_2}{m_1+m_2}\]

\[I=\mu r_0^2\]

\[E_J=\frac{\hbar^2}{2I}J(J+1)\]

\[\omega=\sqrt{\frac{k}{\mu}}\]

\[\Delta E=hf=\hbar\omega\]

T5 — Sólidos

\[E_F=\frac{h^2}{8m}\left(\frac{3N}{\pi V}\right)^{2/3}\]

\[E_{\mathrm{tot}}=\frac35NE_F\]

\[v_F=\sqrt{\frac{2E_F}{m_e}}\]

\[\tau=\frac{m_e}{ne^2\rho}\]

\[\ell=v_F\tau\]

\[v_g=\frac1\hbar\frac{dE}{dk}\]

\[m^*=\frac{\hbar^2}{d^2E/dk^2}\]

T6 — Relatividad

\[\gamma=\frac{1}{\sqrt{1-v^2/c^2}}\]

\[\Delta t=\gamma\Delta\tau\]

\[L=\frac{L_0}{\gamma}\]

\[K=(\gamma-1)mc^2\]

\[E^2=(pc)^2+(mc^2)^2\]

\[N=N_0e^{-t'/\tau}\]

T7 — Nuclear

\[N=N_0e^{-\lambda t}\]

\[T_{1/2}=\frac{\ln2}{\lambda}\]

\[A_{\mathrm{act}}=\lambda N\]

\[E_b=\Delta mc^2\]

\[Q=(m_i-m_f)c^2\]

\[K_\alpha=Q\frac{A-4}{A}\]

T8/T9 — Partículas

\[E^2=(pc)^2+(mc^2)^2\]

\[E_\gamma=pc\]

\[E_{\mathrm{CM,min}}=\sum m_fc^2\]

\[Q=(m_i-m_f)c^2\]
